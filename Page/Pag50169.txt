    PROCEDURE GetFirstDoc@1100281013();
    BEGIN
      FilePath := '';

    END;

    PROCEDURE GetPreviousDoc@1000000003();
    VAR
      i@1100281000 : Integer;
    BEGIN
      FilePath := '';
      FileExtension := '';
      IF ISEMPTY THEN
        EXIT;

      CurrentDoc := GetCurrDocNo;
      IF CurrentDoc = 1 THEN BEGIN
        FINDLAST;
        CurrentDoc := COUNT;
        EXIT;
      END;

      IF CurrentDoc = 2 THEN BEGIN
        FINDFIRST;
        CurrentDoc := 1;
        EXIT;
      END;

      IF NOT FINDFIRST THEN
        EXIT;

      NEXT(CurrentDoc - 2);
      CurrentDoc := CurrentDoc - 1;
    END;

    PROCEDURE GetNextDoc@1000000002();
    BEGIN
      FilePath := '';
      FileExtension := '';
      IF ISEMPTY THEN
        EXIT;
      IF NEXT = 0 THEN BEGIN
        FINDFIRST;
        CurrentDoc := 1;
      END ELSE
        CurrentDoc += 1;
    END;

      lDocAttach@1100281000 : Record 50021;

      lDocAttach.COPY(Rec);

    END;
