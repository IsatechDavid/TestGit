    LOCAL PROCEDURE NoOnFormat@19031395();

        "No.Emphasize" := TRUE;

    LOCAL PROCEDURE NameOnFormat@19039177();
    BEGIN
      IF Type = Type::Company THEN
        NameEmphasize := TRUE
      ELSE BEGIN
        Cont.SETCURRENTKEY("Company Name","Company No.",Type,Name);
        IF ("Company No." <> '') AND (NOT HASFILTER) AND (NOT MARKEDONLY) AND (CURRENTKEY = Cont.CURRENTKEY)
        THEN
          NameIndent := 1;
      END;
    END;

    PROCEDURE CheckFilters@1100281000();
    VAR
      lCorpFilter@1100281002 : Code[20];
    BEGIN
      IF GETFILTER("Corporate No.") <> '' THEN BEGIN
        lCorpFilter := GETFILTER("Corporate No.");
        lCorpFilter :=DELCHR(lCorpFilter, '=', '@*');
        IF Corporate."No." <> lCorpFilter THEN BEGIN
          IF STRLEN(lCorpFilter) > MAXSTRLEN(Corporate."No.") THEN BEGIN
            MESSAGE(Text001, lCorpFilter);
            lCorpFilter := '';
          END
          ELSE
            IF NOT Corporate.GET(GETFILTER("Corporate No.")) THEN BEGIN
              MESSAGE(Text001, lCorpFilter);
              lCorpFilter := '';
            END;
        END;
        IF lCorpFilter = '' THEN
          SETRANGE("Corporate No.")
        ELSE
          SETRANGE("Corporate No.", lCorpFilter);
      END;
    END;

    PROCEDURE LoadList@1100281001(pCheckFilyerChange@1100281003 : Boolean);
    VAR
      lTempCont@1100281000 : TEMPORARY Record 5050;
      lCont@1100281001 : Record 5050;
      lInt@1100281002 : Integer;
    BEGIN
      IF ContFilters.GETFILTERS = GETFILTERS THEN
        EXIT;
      lTempCont := Rec;
      RESET;
      DELETEALL;
      FOR lInt := 0 TO 5 DO BEGIN
        FILTERGROUP(lInt);

      ContFilters.COPY(Rec);
    END;

    PROCEDURE CopyContacts@1100281003(VAR pFromContact@1100281000 : Record 5050) rPartialResult : Boolean;
    VAR
      lWindow@1100281002 : Dialog;
      lInt@1100281001 : Integer;
      lContacts@1100281003 : Integer;
    BEGIN
      lWindow.OPEN(Text002, lInt);
      IF pFromContact.FIND('-') THEN
        REPEAT
          Rec := pFromContact;
          INSERT(FALSE);
          lInt += 1;
          lContacts += 1;
          lWindow.UPDATE(1);
          IF CRMSetup."Contact Search Limit" <> 0 THEN BEGIN
            IF lInt >= CRMSetup."Contact Search Limit" THEN BEGIN
              lInt := 0;
              IF NOT CONFIRM(STRSUBSTNO(Text003, lContacts)) THEN
                EXIT(TRUE);
            END;
          END
          ELSE BEGIN
            IF lContacts > 200 THEN BEGIN
              EXIT(TRUE);
            END;
          END;
        UNTIL pFromContact.NEXT=0;
      lWindow.CLOSE;
      EXIT(FALSE);
    END;
