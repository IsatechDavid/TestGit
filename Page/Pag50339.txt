      TEMP.ADDTEXT('  </br>');

      TEMP.ADDTEXT('</body>');
      TEMP.ADDTEXT('</html>');
      HTML := TEMP;

      IF NOT ISSERVICETIER THEN
        ERASE(lTempFileName);
    END;
