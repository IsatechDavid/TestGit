                     ToJobTaskHeaderNo,
                     AmountAdjustFactor,
                     ApplyPrices,
                     RoundingMethod,
                     ToJobTaskNo,
                     CopyQuantity);

                   NoOfLinesInserted := CopyJob.CopyJob;

                     MESSAGE(Text003,NoOfLinesInserted,ToJobTaskHeaderNo)
                   ELSE
                     MESSAGE(Text005,FromJobTaskHeaderNo);
