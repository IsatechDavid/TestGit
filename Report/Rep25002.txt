    PROCEDURE ProcessTempEntries@1480001(ItemLedgEntry@1480000 : Record 32;Item@1480003 : Record 27;KitItem@1480002 : Boolean);
    VAR
      TempEntryNo@1480001 : Integer;
    BEGIN
      TempValueEntry.RESET;
      IF TempValueEntry.FINDLAST THEN
        TempEntryNo := TempValueEntry."Entry No." + 1
      ELSE
        TempEntryNo := 1;

      TempValueEntry.SETRANGE("Item No.",ItemLedgEntry."Item No.");
      IF KitItem THEN
        TempValueEntry.SETRANGE("Item Charge No.",ItemLedgEntry."Source No.")
      ELSE
        TempValueEntry.SETRANGE("Item Charge No.",'');
      WITH ItemLedgEntry DO BEGIN
        IF NOT TempValueEntry.FINDFIRST THEN BEGIN
          TempValueEntry.INIT;
          TempValueEntry."Entry No." := TempEntryNo;
          TempValueEntry."Item No." := "Item No.";
          TempValueEntry."Posting Date" := "Posting Date";
          TempValueEntry."Item Ledger Entry Type" := "Entry Type"::Sale;
          TempValueEntry."Source No." := "Source No.";
          TempValueEntry."Document No." := "Document No.";
          TempValueEntry.Description := Item.Description;
          TempValueEntry."Location Code" := "Location Code";
          TempValueEntry."Item Ledger Entry No." := "Entry No.";
          TempValueEntry."Item Ledger Entry Quantity" := Quantity;
          TempValueEntry."Invoiced Quantity" := "Invoiced Quantity";
          IF KitItem THEN BEGIN
            TempValueEntry."Item Charge No." := "Source No."
          END ELSE
            TempValueEntry."Item Charge No." := '';
          TempValueEntry.INSERT;
        END ELSE BEGIN
          TempValueEntry."Item Ledger Entry No." := "Entry No.";
          TempValueEntry."Invoiced Quantity" := TempValueEntry."Invoiced Quantity" + "Invoiced Quantity";
          TempValueEntry.MODIFY;
        END;
        TempEntryNo := TempEntryNo + 1;
      END;
    END;

    BEGIN
    END.
