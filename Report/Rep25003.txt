    VAR
      ProductionBOMHeader@1480002 : Record 99000771;
      ProductionBOMLine2@1480003 : Record 99000772;
      Item2@1480012 : Record 27;
      Resource2@1480015 : Record 156;
      ItemFilter@1480001 : Text[250];
      BackOrderQty@1480005 : ARRAY [2] OF Decimal;
      GrossRequirement@1480006 : ARRAY [2] OF Decimal;
      ScheduledReceipt@1480007 : ARRAY [2] OF Decimal;
      PlannedOrderReceipt@1480009 : ARRAY [2] OF Decimal;
      ProjAvailBalance@1480010 : ARRAY [2] OF Decimal;
      TotalKitCost@1480013 : Decimal;
      TotalKitAvail@1480014 : Decimal;

      UsedinOtherKits@1480004 : Boolean;

    PROCEDURE CalcNeed@1(Item@1000 : Record 27;LocationFilter@1001 : Text[250];ReorderPoint@1003 : Decimal;element@1480000 : Integer);
    BEGIN
      WITH Item DO BEGIN
        SETFILTER("Location Filter",LocationFilter);

        SETRANGE("Date Filter",0D,WORKDATE);
        CALCFIELDS(
          "Qty. on Purch. Order",
          "Planning Receipt (Qty.)",
          "Scheduled Receipt (Qty.)",
          "Purch. Req. Receipt (Qty.)",
          "Qty. in Transit",
          "Trans. Ord. Receipt (Qty.)",
          "Reserved Qty. on Inventory");
        BackOrderQty[element] :=
          "Qty. on Purch. Order" + "Scheduled Receipt (Qty.)" +
          "Qty. in Transit" + "Trans. Ord. Receipt (Qty.)" +
          "Planning Receipt (Qty.)" + "Purch. Req. Receipt (Qty.)";

        SETRANGE("Date Filter",0D,31129999D);
        GrossRequirement[element] := AvailToPromise.CalcGrossRequirement(Item) + "Reserved Qty. on Inventory";
        ScheduledReceipt[element] := AvailToPromise.CalcScheduledReceipt(Item);

        CALCFIELDS(
          Inventory,
          "Planning Receipt (Qty.)",
          "Planned Order Receipt (Qty.)",
          "Purch. Req. Receipt (Qty.)");

        PlannedOrderReceipt[element] :=
          "Planning Receipt (Qty.)" +
          "Planned Order Receipt (Qty.)" +
          "Purch. Req. Receipt (Qty.)";

        ProjAvailBalance[element] :=
          Inventory +
          PlannedOrderReceipt[element] +
          ScheduledReceipt[element] -
          GrossRequirement[element];

      END;
    END;

    BEGIN
    END.
