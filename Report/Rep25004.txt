    PROCEDURE ProcessTempEntries@1480001(ItemLedgEntry@1480000 : Record 32;KitItem@1480002 : Boolean;KitItemEntryNo@1480003 : Integer);
    VAR
      KitSalesShipmentLine@1480004 : Record 25004;
      QtyPer@1480005 : Decimal;
    BEGIN
      TempValueEntry.RESET;
      IF TempValueEntry.FINDLAST THEN
        TempEntryNo := TempValueEntry."Entry No." + 1
      ELSE
        TempEntryNo := 1;
      TempValueEntry.SETRANGE("Item No.",ItemLedgEntry."Item No.");

      IF KitItem THEN
        TempValueEntry.SETRANGE("Item Charge No.",ItemLedgEntry."Source No.")
      ELSE
        TempValueEntry.SETRANGE("Item Charge No.",'');

      WITH ItemLedgEntry DO BEGIN
        IF NOT TempValueEntry.FIND('-') THEN BEGIN
          TempValueEntry.INIT;
          TempValueEntry."Entry No." := TempEntryNo;
          TempValueEntry."Item No." := "Item No.";
          TempValueEntry."Posting Date" := "Posting Date";
          TempValueEntry."Item Ledger Entry Type" := "Entry Type";
          TempValueEntry."Source No." := "Source No.";
          TempValueEntry."Document No." := "Document No.";
          IF Item4.GET("Item No.") THEN
            TempValueEntry.Description := Item4.Description;
          TempValueEntry."Document Type" := "Document Type";
          TempValueEntry."Document Line No." := "Document Line No.";
          TempValueEntry."Applies-to Entry" := KitItemEntryNo;
          TempValueEntry."Location Code" := "Location Code";
          TempValueEntry."Item Ledger Entry No." := "Entry No.";
          IF KitItem THEN BEGIN
            KitSalesShipmentLine.SETRANGE("Document No.","Document No.");
            KitSalesShipmentLine.SETRANGE("Document Line No.","Document Line No.");
            KitSalesShipmentLine.SETRANGE(Type,KitSalesShipmentLine.Type::Item);
            KitSalesShipmentLine.SETRANGE("No.","Item No.");
            IF KitSalesShipmentLine.FINDFIRST THEN BEGIN
              QtyPer := KitSalesShipmentLine."Quantity per";
            END ELSE BEGIN
              ProdBOMLine.RESET;
              ProdBOMLine.SETCURRENTKEY(Type,"No.");
              ProdBOMLine.SETRANGE(Type,ProdBOMLine.Type::Item);
              ProdBOMLine.SETRANGE("No.","Item No.");
              IF Item4.GET("Source No.") AND (Item4."Kit BOM No." <> '') THEN
                ProdBOMLine.SETRANGE("Production BOM No.",Item4."Kit BOM No.");
              IF ProdBOMLine.FINDFIRST THEN
                QtyPer := ProdBOMLine."Quantity per";
            END;
            TempValueEntry."Invoiced Quantity" :=
              "Invoiced Quantity" * QtyPer;
            TempValueEntry."Item Ledger Entry Quantity" := Quantity;
            TempValueEntry."Cost Amount (Actual)" := "Cost Amount (Actual)" * QtyPer;
            TempValueEntry."Sales Amount (Actual)" := "Sales Amount (Actual)" * QtyPer;
            TempValueEntry."Item Charge No." := "Source No."
          END ELSE BEGIN
            TempValueEntry."Item Ledger Entry Quantity" := Quantity;
            TempValueEntry."Cost Amount (Actual)" := "Cost Amount (Actual)";
            TempValueEntry."Sales Amount (Actual)" := "Sales Amount (Actual)";
            TempValueEntry."Invoiced Quantity" := "Invoiced Quantity";
            TempValueEntry."Item Charge No." := '';
          END;
          TempValueEntry.INSERT;
        END ELSE BEGIN
          IF KitItem THEN BEGIN

    PROCEDURE CalcProfitPercent@1480008(TempValueEntry3@1480000 : TEMPORARY Record 5802) : Decimal;
    VAR
      ItemLedgEntry2@1480001 : Record 32;
      TotalCost@1480002 : Decimal;
      TotalSales@1480003 : Decimal;
    BEGIN
      CLEAR(TotalCost);
      CLEAR(TotalSales);
      WITH TempValueEntry3 DO BEGIN
        ItemLedgEntry2.RESET;
        ItemLedgEntry2.SETRANGE("Item No.","Item Charge No.");
        ItemLedgEntry2.SETFILTER("Invoiced Quantity",'<>0');
        IF DateFilter <> '' THEN
          ItemLedgEntry2.SETFILTER("Posting Date",DateFilter);
        ItemLedgEntry2.SETFILTER("Entry Type",'%1',ItemLedgEntry2."Entry Type"::Sale);
        ItemLedgEntry2.SETFILTER("Prod. Order No.",'');
        IF ItemLedgEntry2.FINDFIRST THEN
          REPEAT
            ItemLedgEntry2.CALCFIELDS("Sales Amount (Actual)","Cost Amount (Actual)");
            IF ItemLedgEntry2."Cost Amount (Actual)" <> 0 THEN BEGIN
              TotalCost := TotalCost + ABS(ItemLedgEntry2."Cost Amount (Actual)");
            END ELSE BEGIN
              IF Item4.GET(ItemLedgEntry2."Item No.") THEN BEGIN
                IF Item4."Costing Method" = Item4."Costing Method"::Standard THEN
                  TotalCost := TotalCost +
                    ABS(Item4."Standard Cost" * ItemLedgEntry2."Invoiced Quantity")
                ELSE BEGIN
                  IF Item4."Unit Cost" <> 0 THEN
                    TotalCost := TotalCost +
                      ABS(Item4."Unit Cost" * ItemLedgEntry2."Invoiced Quantity")
                  ELSE
                    TotalCost := TotalCost +
                      ABS(Item4."Standard Cost" * ItemLedgEntry2."Invoiced Quantity");
                END;
              END;
            END;
            TotalSales := TotalSales + ItemLedgEntry2."Sales Amount (Actual)";
          UNTIL ItemLedgEntry2.NEXT = 0;
        IF TotalSales <> 0 THEN
          EXIT((TotalSales - TotalCost) * 100 / TotalSales)
        ELSE
          EXIT(0);
      END;
    END;
