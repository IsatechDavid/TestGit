    VAR
      lServiceOrderType@1100281000 : Record 5903;

      lSalesInvLine@1100281003 : Record 113;
      lServIntervNo@1100281002 : Integer;
      lNbInterv@1100281004 : Integer;
      lNbAllInterv@1100281005 : Integer;
    BEGIN
      //Calcul sur les douze derniers mois

      lServiceOrderType.SETRANGE("Home Service",TRUE);
      REPEAT
        lServIntervLine.RESET;
        lServIntervLine.SETCURRENTKEY("Assigned Date","Provider No.","Intervention Status",Closed,"Multi specialities");
        lServIntervLine.SETRANGE("Assigned Date", CALCDATE('<-1Y>', WORKDATE) + 1, WORKDATE);
        lServIntervLine.SETRANGE("Service Order Type",lServiceOrderType.Code);
        lServIntervLine.SETRANGE("Provider No.",Provider."No.");
        lNbAllInterv += lServIntervLine.COUNT;
        lServIntervLine.SETFILTER(lServIntervLine."Intervention Status",'<=%1',lServIntervLine."Intervention Status"::Closed);
        lNbInterv += lServIntervLine.COUNT;

        IF lServIntervLine.FINDSET THEN
          REPEAT
            HabComm += ROUND((lServIntervLine."Total Work Amt Excl. VAT" * Provider."Percentage Commission") / 100, 0.01);
            HabAmount += lServIntervLine."Total Order Amount Incl. VAT";
          UNTIL lServIntervLine.NEXT = 0;

      UNTIL lServiceOrderType.NEXT = 0;

      IF  lNbAllInterv <> 0 THEN BEGIN
        HabTransRate := 100*(lNbInterv/lNbAllInterv);
        HabTransRate := ROUND(HabTransRate,0.01,'=');
      END
      ELSE
        HabTransRate :=0;

      HabIntervNb += lNbInterv;
    END;
