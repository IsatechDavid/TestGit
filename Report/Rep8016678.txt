                             GenFADeprBook."Depreciation Method"::"DB1/SL" :
                               GenValues[7] := GenFADeprBook.Depreciation * -1;
                           END;
                           GenValues[10] := GenValues[5] + GenValues[6] + GenValues[7];
                           GenValues[11] := GenValues[10];

                             FiscValues[1] := GenFADeprBook."Acquisition Cost";
                             FiscFADeprBook."No. of Depreciation Years" := GenFADeprBook."No. of Depreciation Years";

                             FiscValues[12] := FiscFADeprBook."Book Value";

                             FiscFADeprBook.SETFILTER("FA Posting Date Filter",'..%1',StartingDate-1);
                             FiscFADeprBook.CALCFIELDS(Depreciation);
                             FiscValues[5] := (FiscFADeprBook.Depreciation * -1) - GenValues[5];
                             FiscFADeprBook.SETFILTER("FA Posting Date Filter",'%1..%2',StartingDate,EndingDate);
                             FiscFADeprBook.CALCFIELDS(Depreciation);
                             CASE FiscFADeprBook."Depreciation Method" OF

                             END
                             ELSE BEGIN

                               FiscRate := 0;
                             END;
                           END;
