    VAR
      lSalesCrMemoLine@1100281000 : Record 115;

      //<<IMH03.ISAT.EBA
      TempLine.RESET;

              "Quantity (Base)" := "Quantity (Base)" + lSalesCrMemoLine."Quantity (Base)";

              "Line Amount" := "Line Amount" + lSalesCrMemoLine."Line Amount";
              "Unit Price" := ("Line Discount Amount" + "Line Amount") / Quantity;
              MODIFY;
            END ELSE BEGIN
              TempLine := lSalesCrMemoLine;
              TempLine.INSERT;
            END;
          UNTIL lSalesCrMemoLine.NEXT = 0;

          RESET;
          FINDFIRST;
        END;
      END;
      //>>IMH03.ISAT.EBA
    END;
