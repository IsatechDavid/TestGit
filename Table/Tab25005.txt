    VAR
      SalesInvHeader@1000 : Record 112;
    BEGIN

        EXIT(SalesInvHeader."Currency Code");

    END;

    LOCAL PROCEDURE GetCaptionClass@34(FieldNumber@1000 : Integer) : Text[80];
