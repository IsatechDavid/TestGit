    PROCEDURE CheckManpower@1100281030();
    VAR
      lProviderSpecialization@1100281000 : Record 50006;
      lExitLoop@1100281002 : Boolean;
      lManpowerNb@1100281003 : Integer;
    BEGIN
      // IMH10.PC
      // On test que le nombre n'est pas sup‚rieur … l'effectif global
      // On test si l'effectif IMH n'est pas sup‚rieur … l'effectif de la sp‚cialit‚
      // V‚rifie si la somme des effectifs des sp‚cialit‚s n'est pas sup‚rieur … l'effectif global
      // si l'effectif d‚di‚ … imh est … 0 on copie l'ffectif dans ce champs

      Provider.GET("Provider No.");
      IF Provider.Manpower < "Speciality Manpower" THEN BEGIN
        ERROR(Text004, Provider.Manpower);
      END;
      IF "IMH Speciality Manpower" > "Speciality Manpower" THEN BEGIN
        ERROR(Text006);
      END;

      lExitLoop := FALSE;
      lManpowerNb := 0;
      lProviderSpecialization.SETRANGE("Provider No.","Provider No.");
      lProviderSpecialization.SETFILTER("Specialization Code",'<>%1',"Specialization Code");
      IF lProviderSpecialization.FINDSET THEN BEGIN
        REPEAT
          lManpowerNb += lProviderSpecialization."Speciality Manpower";
          IF (lManpowerNb + "Speciality Manpower") > Provider.Manpower THEN BEGIN
            MESSAGE(Text009);
            lExitLoop := TRUE;
          END;
        UNTIL (lProviderSpecialization.NEXT = 0) OR lExitLoop;
      END;

      IF "IMH Speciality Manpower" = 0 THEN BEGIN
        "IMH Speciality Manpower" := "Speciality Manpower";
      END;
    END;
