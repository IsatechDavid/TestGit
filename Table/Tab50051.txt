    PROCEDURE InitFilters@1100281001(pDateFilter@1100281006 : Text[30]);
    VAR
      lAppMgt@1100281000 : Codeunit 1;
      lTeamSalesperson@1100281002 : Record 5084;
      lSalesperson@1100281001 : Code[10];
      lMyTeam@1100281003 : Code[10];
      lTeamFilter@1100281004 : Text[1024];
      lMyGroup@1100281005 : Code[10];
      lProviderSetup@1100281007 : Record 50011;
    BEGIN
      RESET;
      InitFlowFilters;
      SETRANGE("Ressource Filter", lAppMgt.GetUserCode(2));
      lSalesperson := lAppMgt.GetUserCode(1);
      IF lSalesperson <> '' THEN BEGIN
        SETRANGE("Salesperson Filter", lSalesperson);
        SETFILTER("Salesperson Filter2", '=%1|=%2', lSalesperson, '');  //IMH03.ST
      END;

      lTeamSalesperson.SETCURRENTKEY("Salesperson Code");
      lTeamSalesperson.SETRANGE("Salesperson Code", lSalesperson);
      IF lTeamSalesperson.FINDSET(FALSE) THEN
        REPEAT
          IF lTeamSalesperson.Default THEN BEGIN
            lMyTeam := lTeamSalesperson."Team Code";
            lTeamSalesperson.CALCFIELDS("Team Group");
            lMyGroup := lTeamSalesperson."Team Group";
          END;
          IF lTeamFilter = '' THEN
            lTeamFilter := lTeamSalesperson."Team Code"
          ELSE
            lTeamFilter := lTeamFilter + '|' + lTeamSalesperson."Team Code";
        UNTIL lTeamSalesperson.NEXT=0;
      SETRANGE("My Team Filter", lMyTeam);
      SETFILTER("Teams Filter", lTeamFilter);
      //  IF lMyGroup <> '' THEN
      SETRANGE("Team Group Filter", lMyGroup);

      //<<IMH04.ST
      IF pDateFilter <> '' THEN
        SETFILTER("End Date Filter", pDateFilter);
      //>>IMH04.ST
      IF lProviderSetup.GET THEN
        SETRANGE("Network Search Team Filter", lProviderSetup."Not Found Provider Team Code");
      SetIncomingDocFilter;
    END;

    PROCEDURE SetIncomingDocFilter@1100281003();
    VAR
      lDocFlow@1100281000 : Record 50053;
      lTempTeam@1100281002 : TEMPORARY Record 5083;
      lTextFilter@1100281001 : Text[1024];
    BEGIN
      //<<IMH06.ISAT.EBA
      {
      lDocFlow.SETFILTER("Default Team Code", '<>%1', '');
      IF lDocFlow.FINDSET THEN BEGIN
        REPEAT
          IF NOT lTempTeam.GET(lDocFlow."Default Team Code") THEN BEGIN
            lTempTeam.Code := lDocFlow."Default Team Code";
            lTempTeam.INSERT(FALSE);
          END;
        UNTIL lDocFlow.NEXT = 0;

        REPEAT

        UNTIL lTempTeam.NEXT=0;

      }

      ProviderSetup.GET;
      IF ProviderSetup."GED Team Code" <> '' THEN

      //>>IMH06.ISAT.EBA
    END;

    PROCEDURE AddFilterToFilter@1100281004(pFilterToAdd@1100281000 : Text[30];VAR pNewFilter@1100281001 : Text[1024]);
    BEGIN
      IF pNewFilter = '' THEN BEGIN
        pNewFilter := pFilterToAdd;
        EXIT;
      END;
      IF STRLEN(pFilterToAdd) + STRLEN(pNewFilter) +1 > MAXSTRLEN(pNewFilter) THEN
        ERROR(STRSUBSTNO(Text001, MAXSTRLEN(pNewFilter)))
      ELSE
        pNewFilter := pNewFilter + '|' + pFilterToAdd;
    END;
